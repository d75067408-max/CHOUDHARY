<script>
// 🔥 SETTINGS
const latestVersion = "1.1";
const apkLink = sha256:382f14b40585b675b88b9144fedfc3a104a315f21d2e6f29fd5d200abef56641

// VERSION SYSTEM
let currentVersion = localStorage.getItem("appVersion") || "1.0";
document.getElementById("version").innerText = latestVersion;

if(currentVersion !== latestVersion){
  document.getElementById("updatePopup").style.display = "block";
}

// DOWNLOAD COUNTER
let downloads = localStorage.getItem("downloads") || 0;
document.getElementById("downloads").innerText = downloads;

function downloadApp(){
  downloads++;
  localStorage.setItem("downloads", downloads);
  localStorage.setItem("appVersion", latestVersion);
  window.location.href = apkLink;
}

// REVIEWS SYSTEM
let reviews = JSON.parse(localStorage.getItem("reviews")) || [];

function addReview(){
  let name = document.getElementById("name").value;
  let review = document.getElementById("review").value;

  if(name && review){
    reviews.push({name, review});
    localStorage.setItem("reviews", JSON.stringify(reviews));
    displayReviews();
  }
}

function displayReviews(){
  let html = "";
  reviews.forEach(r => {
    html += `<div class="review-box"><b>${r.name}</b><br>${r.review}</div>`;
  });
  document.getElementById("reviews").innerHTML = html;
}

displayReviews();

// DOWNLOAD LINK
document.getElementById("downloadLink").href = apkLink;

</script>
